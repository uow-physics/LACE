63 void LpcTextOutput::storeCurve(const LpcCurve* mainCurve)
64 {
65 
66  // Store the main curve and all of its branches
67  if (!mainCurve) {return;}
68 
69  int curveId = mainCurve->getIndex();
70  writeData_ << "MainCurve:Id " << curveId << std::endl;
71  this->storeCurveDetails(mainCurve);
72 
73  // Retrieve the branches
74  LpcBranchCollection* theBranches = mainCurve->getBranchCollection();
75 
76  if (theBranches) {
77 
78  // Get the number of generational levels for the branches
79  int nLevels = theBranches->getNumberLevels();
80 
81  // Loop over each branch generation
82  for (int i = 0; i < nLevels; i++) {
83 
84  int iG = i + 1; // Generation number
85  // Get the vector of branches
86  std::vector<LpcBranch*> branchVect = theBranches->getBranches(iG);
87 
88  // Loop over the branches
89  std::vector<LpcBranch*>::const_iterator iter;
90  for (iter = branchVect.begin(); iter != branchVect.end(); ++iter) {
91 
92  const LpcBranch* theBranch = *iter;
93  // Write out the information stored in the branch
94  if (theBranch) {
95 
96  writeData_ << "Branch:Id " << theBranch->getIndex() << std::endl;
97  writeData_ << "Branch:MainCurveId " << curveId << std::endl;
98  writeData_ << "Branch:GenerationLevel " << iG << std::endl;
99 
100  this->storeCurveDetails(theBranch);
101 
102  } // Check if branch pointer exists
103 
104  } // Iteration over branches
105 
106  } // Branch generation loop
107 
108  } // Check for any stored branches

112 void LpcTextOutput::storeCurveDetails(const LpcAbsCurve* theCurve)
113 {
114 
115  // Store either the main curve or branch, following the info in
116  // LpcAbsCurve::print()
117 
118  if (!theCurve) {return;}
119 
120  // Starting point
121  LpcPoint startPoint = theCurve->getStartPoint();
122  this->printPoint("LpcStartPoint", startPoint);
123 
124  // LpcPoints
125  std::vector<LpcPoint> lpcPoints = theCurve->getLpcPoints();
126  int nLpc = lpcPoints.size();
127  writeData_ << "NumberOfLpcPoints " << nLpc << std::endl;
128 
129  int i(0);
130 
131  for (i = 0; i < nLpc; i++) {
132  std::string word("LpcPoint");
133  this->printPoint(word, i, lpcPoints[i]);
134  }
135 
136  // Eigenvectors
137  Eigen::MatrixXd eigenVectors = theCurve->getEigenVectors();
138  this->printMatrix("LpcEigenVectors", eigenVectors);
139 
140  // Cosine angles
141  Eigen::VectorXd cosAngles = theCurve->getCosAngles();
142  this->printVector("LpcCosineAngles", cosAngles);
143 
144  // Rho
145  Eigen::VectorXd rho = theCurve->getRho();
146  this->printVector("EigenRatioRho", rho);
147 
148  // c0
149  Eigen::VectorXd c0 = theCurve->getc0();
150  this->printVector("c0", c0);
151 
152  // HighRhoPoints
153  std::vector<LpcPoint> highRhoPoints = theCurve->getHighRhoPoints();
154  int nHRho = highRhoPoints.size();
155 
156  writeData_ << "HighRhoPoints " << nHRho << std::endl;
157  for (i = 0; i < nHRho; i++) {
158  std::string word("HighRhoPoint");
159  this->printPoint(word, i, highRhoPoints[i]);
160  }
161 
162 
163  // Pathlengths object
164  LpcPathLength pathLength = theCurve->getPathLength();
165 
166  // Lambda
167  Eigen::VectorXd lambda = pathLength.getLambda();
168  this->printVector("LpcLambda", lambda);
169 
170  // Lambda axes
171  Eigen::MatrixXd lambdaAxes = pathLength.getLambdaAxes();
172  this->printMatrix("LpcLambdaAxes", lambdaAxes);
173 
174  // Delta lambda
175  Eigen::VectorXd deltaLambda = pathLength.getDeltaLambda();
176  this->printVector("LpcPathLength", deltaLambda);
177 
178  // Residuals object
179  LpcResiduals residuals = theCurve->getResiduals();
180 
181  // Lpc point residuals averaged over all nearest hits
182  Eigen::VectorXd lpcRes = residuals.getLpcResiduals();
183  this->printVector("MeanLpcResiduals", lpcRes);
184 
185  // Hit-to-nearest-lpc point residuals
186  Eigen::VectorXd hitRes = residuals.getHitResiduals();
187  this->printVector("HitResiduals", hitRes);
188 
189  // Weighted lpc point residuals averaged over all nearest hits
190  Eigen::VectorXd wLpcRes = residuals.getWeightedLpcResiduals();
191  this->printVector("WeightMeanLpcResiduals", wLpcRes);
192 
193  // Weighted hit-to-nearest-lpc point residuals
194  Eigen::VectorXd wHitRes = residuals.getWeightedHitResiduals();
195  this->printVector("WeightHitResiduals", wHitRes);
196 
197  // The index number of the nearest lpc point for each hit
198  Eigen::VectorXi hitNearLpc = residuals.getHitNearestLpc();
199  this->printVector("HitNearestLpc", hitNearLpc);
200 
201  // Feature points (cosine peaks)
202  std::vector<int> cosPeakIndices = theCurve->getCosPeakIndices();
203  this->printVector("CosPeaks", cosPeakIndices);
204
