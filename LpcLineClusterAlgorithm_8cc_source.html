74 {
75 
76  // Check if the curve pointer is valid
77  if (!theCurve) {return LpcClusterData();}
78 
79  // The hits are always the same for the main curve or its branches
80  theHits_ = theCurve->getHitCollection();
81 
82  // The unscaled hit co-ordinates and number of co-ordinate dimensions
83  if (theHits_) {
84  hitPoints_ = theHits_->getCoords();
85  nHitPoints_ = theHits_->getNumberOfHits();
86  nDim_ = theHits_->getNDimensions();
87  }
88 
89  // An array to keep track of which hits have already been used for clusters
90  usedHits_ = Eigen::VectorXi::Zero(nHitPoints_);
91 
92  LpcClusterData mainResults = this->processCurve(theCurve);
93 
94  LpcBranchCollection* branchCollection = theCurve->getBranchCollection();
95 
96  if (branchCollection && doBranchVtx_) {
97 
98  int nLevels = branchCollection->getNumberLevels();
99 
100  // Loop over the number of generations (starts at 1 not 0)
101  for (int iL = 1; iL <= nLevels; iL++) {
102 
103  std::vector<LpcBranch*> branches = branchCollection->getBranches(iL);
104  std::vector<LpcBranch*>::const_iterator bIter;
105  for (bIter = branches.begin(); bIter != branches.end(); ++bIter) {
106 
107  const LpcBranch* theBranch = *bIter;
108 
109  if (!theBranch) {continue;}
110 
111  LpcClusterData branchResults = this->processCurve(theBranch);
112  // Add the branch results to the main results
113  mainResults.addData(branchResults);
114 
115  }
116 
117  }
118 
119  }
120 
121  // Check to see if there are any clusters. If not, then create a single
122  // all-encompassing cluster from the list of hits
123  int nClusters = mainResults.getNumberClusters();
124 
125  if (nClusters < 1) {
126 
127  int clusterIndex(0);
128  LpcCluster* oneCluster = this->createSingleCluster(clusterIndex);
129  // Specify the vertex as the position of the hit with the lowest
130  // value of the first co-ordinate ("x")
131  LpcVertex* oneVertex = this->createSingleVertex();
132  mainResults.addCluster(oneCluster);
133  mainResults.addVertex(oneVertex);
134 
135  }
136

145 LpcClusterData
146 LpcLineClusterAlgorithm::processCurve(const LpcAbsCurve* theCurve)
147 {
148 
149  LpcClusterData theResults;
150  if (!theCurve) {return theResults;}
151 
152  int curveIndex = theCurve->getIndex();
153  int curveFlag = theCurve->getFlag();
154  curveId_ = curveIndex;
155  branchId_ = 0;
156 
157  if (curveFlag > -1) {
158  // We have a branch. Set the curveId to the main curve index number,
159  // which is equal to the flag value, and set the branchId to the
160  // index number of the branch
161  curveId_ = curveFlag;
162  branchId_ = curveIndex;
163  }
164 
165  // The vector of lpc points
166  lpcPoints_ = theCurve->getLpcPoints();
167  nLpcPoints_ = theCurve->getNLpcPoints();
168  nLpcPoints1_ = nLpcPoints_ - 1;
169 
170  // The lpc residuals
171  lpcResiduals_ = theCurve->getResiduals();
172 
173  // Get the Eigen vector containing the lpc residual distances for each hit
174  hitResiduals_ = lpcResiduals_.getHitResiduals();
175 
176  // Get the Eigen vector containing the closest lpc point integer for each hit
177  hitNearLpc_ = lpcResiduals_.getHitNearestLpc();
178 
179  // Get the cosine peak ranges for the feature points
180  cosPeakRanges_ = theCurve->getCosPeakRanges();
181  nPeaks_ = cosPeakRanges_.size();
182 
183  // If there are no peaks, stop
184  if (nPeaks_ < 1) {return theResults;}
185 
186  // Find the lpc ranges that straddle the lpc feature points
187  this->findLpcRanges();
188 
189  // Find the line segments that are near the feature points
190  this->findLineRanges();
191  this->findLineSegments();
192 
193  // Find the initial vertices. Matrix has row = vertex, col = co-ordinates (x,y,z)
194  Eigen::MatrixXd initVtxPoints = this->findVtxPoints(lineSegments_);
195 
196  // Extend the line segments, using the initial vertex position
197  this->findExtendedLines(initVtxPoints);
198 
199  // Find the updated vertex positions
200  Eigen::MatrixXd extVtxPoints = this->findVtxPoints(extLineSegments_);
201 
202  // Create clusters based on the associated hits, together with
203  // the nearest hits along the rest of the lpc point range
204  std::vector<LpcCluster*> lpcClusters = this->formClusters(extVtxPoints);
205 
206  // Add any left over hits to the (nearest) cluster(s)
207  this->addRemainingHits(lpcClusters);
208 
209  // Create the object storing the cluster and vertex pointer information
210  LpcClusterData mergedVtxClusters = this->mergeClusters(extVtxPoints, lpcClusters);
211 
212  if (mergedOK_) {
213  // We have created a new, merged set of LpcCluster pointers.
214  // Delete the original LpcCluster pointers
215  std::vector<LpcCluster*>::iterator cIter;
216  for (cIter = lpcClusters.begin(); cIter != lpcClusters.end(); ++cIter) {
217  delete (*cIter);
218  }
219  }
220 
221  // Return the merged clusters and vertices
222  return mergedVtxClusters;
223

227 {
228 
229  // Find the ranges of the lpc points that straddle the feature points
230  // Clear the internal vector of lpc ranges
231  lpcRanges_.clear();
232 
233  if (nPeaks_ > 0) {
234 
235  LpcBinRange firstRange = cosPeakRanges_[0];
236  int firstMinBin = firstRange.getMinBin();

238 
239  lpcRanges_.push_back(LpcBinRange(0, firstMinBin));
240 
241  }
242

244 
245  lpcRanges_.push_back(LpcBinRange(firstRange.getMaxBin(), nLpcPoints1_));
246 
247  } else {
248 
249  LpcBinRange prevRange = firstRange;
250  LpcBinRange nextRange;
251 
252  for (int iP = 1; iP < nPeaks_; iP++) {
253  nextRange = cosPeakRanges_[iP];
254  lpcRanges_.push_back(LpcBinRange(prevRange.getMaxBin(), nextRange.getMinBin()));
255  prevRange = nextRange;
256  }
257 
258  int nextMaxBin = nextRange.getMaxBin();
259  if (nextMaxBin < nLpcPoints1_) {
260  lpcRanges_.push_back(LpcBinRange(nextMaxBin, nLpcPoints1_));
261  }
262 
263  } // nPeaks > 1
264 
265  } // nPeaks > 0

269 void LpcLineClusterAlgorithm::findLineRanges()
270 {
271  // For each feature point, find the lpc point range numbers that are "nLpcSegment"
272  // lpc points away. This allows us to define line segments for vertex finding.
273  // The cosine peak ranges should already be ordered.
274  lineRanges_.clear();
275 
276  for (int iP = 0; iP < nPeaks_; iP++) {
277 
278  LpcBinRange peakRange = cosPeakRanges_[iP];
279 
280  int minBin = peakRange.getMinBin();
281  int maxBin = peakRange.getMaxBin();
282 
283  int lowP1 = this->checkLpcInt(minBin - nLpcSegment_);
284  int highP1 = this->checkLpcInt(minBin);
285 
286  int lowP2 = this->checkLpcInt(maxBin);
287  int highP2 = this->checkLpcInt(maxBin + nLpcSegment_);
288 
289  lineRanges_.push_back(LpcBinRange(lowP1, highP1));
290  lineRanges_.push_back(LpcBinRange(lowP2, highP2));
291 
292  }

296 void LpcLineClusterAlgorithm::findLineSegments()
297 {
298 
299  // Find the straight line segments that pass through each
300  // of the line ranges that are next to the lpc feature points.
301  lineSegments_.clear();
302 
303  // For each lpc point range, retrieve the closest set of data hits
304  // that have residuals below a maximum threshold.
305  // Then, find the centroid position and fit a best straight line
306  // segment to these same data hits
307 
308  for (size_t iR = 0; iR < lineRanges_.size(); iR++) {
309 
310  // Lpc point line range
311  LpcBinRange theRange = lineRanges_[iR];
312  int startRange = theRange.getMinBin();
313  int endRange = theRange.getMaxBin();
314  int nRange = endRange - startRange + 1;
315 
316  // Keep track of the hit positions that are closest to the lpc
317  // curve "line segment". Also store the lpc-to-hit residuals
318 
319  // Create vector containers for the hits that are closest to each lpc point
320  std::vector<int> matchedHitInt;
321 
322  // Loop over the lpc point integers in this line range
323  for (int iL = 0; iL < nRange; iL++) {
324 
325  // Lpc point integer
326  int iLpc = startRange + iL;
327 
328  // Loop over the data hits
329  for (int iH = 0; iH < nHitPoints_; iH++) {
330 
331  // Get the integer of the closest lpc point for this hit
332  int iHitLpc = hitNearLpc_(iH);
333 
334  // See if the lpc point integers match
335  if (iHitLpc == iLpc) {
336 
337  // Store the hit number
338  matchedHitInt.push_back(iH);
339 
340  }
341 
342  } // Loop over data hits
343 
344 
345  } // Loop over the lpc points within the given line range
346 
347  // Create an Eigen vector of the matched hit residuals
348  int nMatchedHits = matchedHitInt.size();
349  Eigen::VectorXd matchedHitRes = Eigen::VectorXd::Zero(nMatchedHits);
350 
351  for (int jH = 0; jH < nMatchedHits; jH++) {
352 
353  int hitIndex = matchedHitInt[jH];
354  matchedHitRes(jH) = hitResiduals_(hitIndex);
355 
356  }
357 
358  // Now find the residual average and its root-mean-square (standard deviation)
359  std::pair<double, double> resMeanAndRms = functions_.getMeanAndRms(matchedHitRes);
360  double residualMean = resMeanAndRms.first;
361  double residualRms = resMeanAndRms.second;
362 
363  // Define the maximum selection requirement on the hit residuals
364  double residualCut = this->getResidualCut(residualMean, residualRms);
365 
366  // Loop over the hits and keep those with the required residuals
367  std::vector<int> keptHitIndices;
368  for (int jH = 0; jH < nMatchedHits; jH++) {
369 
370  int hitInt = matchedHitInt[jH];
371  double hitRes = matchedHitRes(jH);
372  if (hitRes < residualCut) {
373  keptHitIndices.push_back(hitInt);
374  }
375 
376  }
377 
378  int nKeptHits = keptHitIndices.size();
379  // Define an Eigen MatrixXd that stores the kept hit coords; row = hit, col = x,y,z,..
380  Eigen::MatrixXd matchedHitPos = Eigen::MatrixXd::Zero(nKeptHits, nDim_);
381  for (int jH = 0; jH < nKeptHits; jH++) {
382 
383  int hitInt = keptHitIndices[jH];
384  matchedHitPos.row(jH) = hitPoints_.row(hitInt);
385 
386  }
387 
388  // Check to see if there are no kept hits. In this case, just use
389  // the lpc points directly
390  if (nKeptHits == 0) {
391 
392  // Re-initalise the matchedHitPos matrix with updated dimensions
393  matchedHitPos = Eigen::MatrixXd::Zero(nRange, nDim_);
394 
395  for (int iL = 0; iL < nRange; iL++) {
396 
397  // Lpc point integer
398  int iLpc = startRange + iL;
399  LpcPoint lpcPoint = lpcPoints_[iLpc];
400  matchedHitPos.row(iL) = lpcPoint.getCoords();
401 
402  }
403 
404  }
405 
406  // Find the centroid and unit direction vector for the straight line section
407  // that minimises the perpendicular Euclidean distance to each hit
408  std::pair<Eigen::VectorXd, Eigen::VectorXd> theLine = lineFitter_.findLine(matchedHitPos);
409 
410  // Store the line segment information
411  if (residualCut < minVtxResCut_) {residualCut = minVtxResCut_;}
412  LpcLineSegment theSegment(theLine.first, theLine.second, residualMean,
413  residualRms, residualCut);
414  lineSegments_.push_back(theSegment);
415 
416  } // Loop over the line ranges
417 
418 }

420 Eigen::MatrixXd
421 LpcLineClusterAlgorithm::findVtxPoints(const std::vector<LpcLineSegment>& lineSegments) const
422 {
423 
424  // Find the point of closest approach between neighbouring ling segments,
425  // using linear vector algebra.
426 
427  // Loop over the line segments
428  int nLines = lineSegments.size();
429  int nVtx = nLines/2;
430  double small(1e-10);
431 
432  // Define a matrix of the vertex points: row = vertex, col = co-ordinates (x,y,z)
433  Eigen::MatrixXd theVtxPoints = Eigen::MatrixXd::Zero(nVtx, nDim_);
434 
435  for (int iL = 0; iL < nVtx; iL++) {
436 
437  int iL2 = 2*iL;
438  LpcLineSegment line1 = lineSegments[iL2];
439  LpcLineSegment line2 = lineSegments[iL2+1];
440 
441  Eigen::VectorXd P0 = line1.getCentroid();
442  Eigen::VectorXd Q0 = line2.getCentroid();
443  Eigen::VectorXd u = line1.getDirection();
444  Eigen::VectorXd v = line2.getDirection();
445 
446  // Various dot products
447  Eigen::VectorXd w0 = P0 - Q0;
448  double a = u.dot(u);
449  double b = u.dot(v);
450  double c = v.dot(v);
451  double d = u.dot(w0);
452  double e = v.dot(w0);
453 
454  // Initialise the lengths along the 2 lines which will be "optimised"
455  // to find the closest distance of approach
456  double sc(0.0), tc(0.0);
457 
458  // Denominator factor for sc and tc terms
459  double D = a*c - b*b;
460 
461  // First check if the two lines are parallel
462  if (D < small) {
463 
464  if (b > c && fabs(b) > small) {
465  tc = d/b;
466  } else if (fabs(c) > small) {
467  tc = e/c;
468  }
469 
470  } else {
471 
472  sc = (b*e - c*d)/D;
473  tc = (a*e - b*d)/D;
474 
475  }
476 
477  // Find the point of closest approach
478  Eigen::VectorXd D0 = P0 + Q0;
479  Eigen::VectorXd du = sc*u;
480  Eigen::VectorXd dv = tc*v;
481  Eigen::VectorXd dL = du + dv;
482 
483  Eigen::VectorXd vtx = 0.5*(D0 + dL);
484 
485  theVtxPoints.row(iL) = vtx;
486 
487  }
488 
489  return theVtxPoints;
490

493 void LpcLineClusterAlgorithm::findExtendedLines(const Eigen::MatrixXd& initVtxPoints)
494 {
495 
496  // Extend the line segments by "hoovering up" hits that lie inside
497  // cylinders whose axes are defined by the initial line segments
498  // (lineSegments_ containing the centroids and direction unit vectors)
499 
500  // Clear the vector of extended line segments
501  extLineSegments_.clear();
502 
503  // Set the end-points of the given line segments as the initial
504  // vertex positions. This will automatically change the sign of the
505  // direction vectors to point to these vertices
506 
507  int nSegments = lineSegments_.size();
508  int iL(0);
509  for (iL = 0; iL < nSegments; iL++) {
510 
511  Eigen::VectorXd V0 = initVtxPoints.row(iL/2);
512  lineSegments_[iL].setEndPoint(V0);
513 
514  }
515 
516  // Find all hits that are associated to the initial line segments.
517  // This is a dictionary where key = line segment number and the
518  // value is a vector of integers of the matched hit indices
519  bool uniqueHits(false);
520  LpcLineIntMap hitMatchDict = this->findAssociatedHits(lineSegments_, uniqueHits);
521 
522  // Loop over the initial line segments and obtain new line fits
523  // that use the extended set of hits
524  for (iL = 0; iL < nSegments; iL++) {
525 
526  LpcLineSegment theLine = lineSegments_[iL];
527 
528  // All matched hits for the extended line
529  std::vector<int> extMatchedHits = hitMatchDict[iL];
530  int nExtHits = extMatchedHits.size();
531 
532  if (nExtHits > 0) {
533 
534  // Define an Eigen MatrixXd that stores the required hit coords; row = hit, col = x,y,z,..
535  Eigen::MatrixXd extHitPositions = Eigen::MatrixXd::Zero(nExtHits, nDim_);
536  for (int kH = 0; kH < nExtHits; kH++) {
537 
538  int hitInt = extMatchedHits[kH];
539  extHitPositions.row(kH) = hitPoints_.row(hitInt);
540 
541  }
542 
543  // Find the centroid and unit direction vector for the extended straight line section
544  // that minimises the perpendicular Euclidean distance to each hit
545  std::pair<Eigen::VectorXd, Eigen::VectorXd> extLine = lineFitter_.findLine(extHitPositions);
546 
547  LpcLineSegment extSegment(extLine.first, extLine.second, theLine.getResidualMean(),
548  theLine.getResidualRms(), theLine.getResidualCut());
549 
550  extLineSegments_.push_back(extSegment);
551 
552 
553  } else {
554 
555  // Keep the original line
556  extLineSegments_.push_back(theLine);
557 
558  }
559 
560  } // Loop over initial line segments

564 LpcLineClusterAlgorithm::LpcLineIntMap
565 LpcLineClusterAlgorithm::findAssociatedHits(const std::vector<LpcLineSegment>& lineSegments,
566  bool uniqueHits)
567 {
568 
569  LpcLineIntMap hitMatchDict;
570 
571  // The number of line segments we have
572  int nLines = lineSegments.size();
573 
574  // Create Eigen objects to store the centroids, direction vectors
575  // and residual cuts, which are needed as we loop over all hits (we want
576  // to avoid re-obtaining these as we iterate through the hit loop)
577  Eigen::MatrixXd centroids = Eigen::MatrixXd::Zero(nLines, nDim_);
578  Eigen::MatrixXd nVectors = Eigen::MatrixXd::Zero(nLines, nDim_);
579  Eigen::VectorXd residualCuts = Eigen::VectorXd::Zero(nLines);
580 
581  // Store the dot product of (V - C)dot n, where V = end point (vertex),
582  // C = centroid, n = direction vector
583  Eigen::VectorXd VCnArray = Eigen::VectorXd::Zero(nLines);
584 
585  // Retrieve the centroid and direction vector for the line segments.
586  // Reverse the direction vector if it is not pointing towards its
587  // corresponding endpoint/vertex position
588 
589  for (int jL = 0; jL < nLines; jL++) {
590 
591  LpcLineSegment lSegment = lineSegments[jL];
592  Eigen::VectorXd C = lSegment.getCentroid();
593  Eigen::VectorXd n = lSegment.getDirection();
594  Eigen::VectorXd V = lSegment.getEndPoint();
595 
596  Eigen::VectorXd VmC = V - C;
597  double VCn = VmC.dot(n);
598 
599  centroids.row(jL) = C;
600  if (VCn < 0.0) {
601  // Reverse the direction vector
602  nVectors.row(jL) = -n;
603  VCnArray(jL) = -VCn;
604  } else {
605  nVectors.row(jL) = n;
606  VCnArray(jL) = VCn;
607  }
608 
609  residualCuts(jL) = lSegment.getResidualCut();
610 
611  }
612 
613  // Loop over all hits and find those that can extend the initial line segments
614  for (int iH = 0; iH < nHitPoints_; iH++) {
615 
616  // Get the hit co-ordinates
617  Eigen::VectorXd P = hitPoints_.row(iH);
618 
619  // Find out which line segment this hit is closest to
620  // (within its residual cut)
621 
622  int matchedLine(-1);
623  double minDelta(1e10);
624 
625  // Loop over the line segments
626  for (int iL = 0; iL < nLines; iL++) {
627 
628  // Get the centroid and direction vector of the line segment
629  Eigen::VectorXd C = centroids.row(iL);
630  Eigen::VectorXd n = nVectors.row(iL);
631  double VCn = VCnArray(iL);
632 
633  // Perpendicular distance of the hit from the line segment.
634  // This makes sure that the point P is pointing towards the end point
635  // of the line V (in the direction of n), but does not go beyond V, i.e.
636  // the distance PC is less than VC
637  double delta = functions_.getPerpLineDist(P, C, n, VCn);
638 
639  // The cut on the residuals
640  double resCut = residualCuts(iL);
641 
642  // Check if the hit is close to the line, and update the minimum delta so far
643  if (delta > -1.0 && delta < resCut && delta < minDelta) {
644  matchedLine = iL;
645  minDelta = delta;
646 
647  }
648 
649  } // Loop over line segments
650 
651  // Append the hit integer for the given line segment
652  if (matchedLine != -1) {
653 
654  hitMatchDict[matchedLine].push_back(iH);
655  if (uniqueHits) {
656  usedHits_(iH) = 1;
657  }
658 
659  }
660 
661  } // Loop over data hits
662 
663 
664  return hitMatchDict;
665

668 std::vector<LpcCluster*> LpcLineClusterAlgorithm::formClusters(const Eigen::MatrixXd& vtxMatrix)
669 {
670 
671  // Form clusters from the hits associated to the various
672  // lpc curve sections separated by feature points/vertices
673  std::vector<LpcCluster*> theClusters;
674 
675  // Set the end points for the extended lines as the new vertex positions
676  int iL(0);
677  int nExtLines = extLineSegments_.size();
678  for (iL = 0; iL < nExtLines; iL++) {
679 
680  Eigen::VectorXd V = vtxMatrix.row(iL/2);
681  extLineSegments_[iL].setEndPoint(V);
682 
683  }
684 
685  // Find the matched hits for the extended line segment
686  bool uniqueHits(true);
687  LpcLineIntMap hitMatchDict1 = this->findAssociatedHits(extLineSegments_, uniqueHits);
688
